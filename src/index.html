	const server = require("fb-server");

	let staticConfig = [
		{path: "dist", alias: "/abc"},
		{path: "src"}
	];


	let proxyConfig = {
		proxyTable: {
			'/api': {
				target: 'http://localhost:8000',
				changeOrigin: false,
				logs: true,
			},
			'/api/v2': {
				target: 'http://localhost:8080',
				changeOrigin: false,
				logs: true,
			},
			"/planb/": {
				target: "http://192.168.0.103:8090/",
				changeOrigin: false,
				logs: false,
			}
		}

	
	};

	let app = server(3000, staticConfig, proxyConfig);
